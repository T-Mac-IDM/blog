1.function [L U]=slu(A)
2.  [n n]= size(A);
3.  tol = 1e-6;
4.  for pivot = 1:n
5.    if abs(A(pivot,pivot)) < tol
6.      disp([num2str(pivot),"th pivot is zero"]) 
7.      return
8.      end
9.    L(pivot,pivot) = 1;
10.        for row = pivot+1:n
11.            L(row,pivot) = A(row,pivot)/A(pivot,pivot);
12.            for col = 1:n
13.                A(row,col) -= A(pivot,col)*L(row,pivot);
14.            end
15.        end
16.    U(pivot,:) = A(pivot,:);
17.    end
1.function x=slv(A,b)
2.  [L U] = slu(A);
3.  [n n] = size(A);
4.  s = 0;
5.  for row=1:n
6.    for col=1:row-1
7.      s += L(row,col)*c(col);
8.    end
9.    c(row) = b(row) - s;
10.  end
11.  s= 0;
12.  for row = n:-1:1
13.    for col = row+1:n
14.      s += U(row,col)*x(col);
15.     end
16.    x(row) = (c(row) - s)/U(row,row);
17.  end
18.  x = x';
